//@version=5
indicator("HaoNguyen Boxes - Ultimate Master V9.5.5 (Fixed Full)", overlay=true, max_boxes_count=500, max_labels_count=500, max_lines_count=500)

// --- [1] INPUTS ---
offset          = input.int(3, "Tương lai Offset", minval=0)
emaLen          = input.int(21, "EMA 21 (Trend)", minval=1)
ma9Len          = input.int(9, "MA 9 (Signal)", minval=1)

showWyckoff     = input.bool(true, "Hiển thị nhãn Wyckoff")
useAutoLookback = input.bool(true, "Tự động tính Lookback")
manualLookback  = input.int(30, "Lookback thủ công", minval=10)
adxThreshold    = input.int(25, "Ngưỡng ADX nhận diện Sideway", minval=15)

volSensitivity  = input.float(1.1, "Độ nhạy Vol", minval=0.5, step=0.1)
vcpRatio        = input.float(0.5, "Ngưỡng VCP", minval=0.1, step=0.05)
showFVG         = input.bool(true, "Hiển thị FVG (ICT)")
maxFVGCount     = input.int(15, "Số lượng FVG tối đa", minval=1, maxval=50)

// Trendlines Settings
grp_trend       = "Auto Trendlines (Finviz Style)"
showTrend       = input.bool(true, "Bật/Tắt Trendline Chính (Đỏ)", group=grp_trend)
pivotLen        = input.int(10, "Độ nhạy Pivot Chính", minval=2, group=grp_trend)
resColor        = input.color(color.red, "Màu Kháng cự Chính", group=grp_trend)
supColor        = input.color(color.green, "Màu Hỗ trợ", group=grp_trend)

// Mới: Cấu hình đường màu tím
showTrendShort  = input.bool(true, "Bật Trendline Ngắn hạn (Tím - Kháng cự)", group=grp_trend)
pivotLenShort   = input.int(4, "Độ nhạy Pivot Ngắn hạn", minval=2, group=grp_trend)
resColorShort   = input.color(color.purple, "Màu Kháng cự Ngắn hạn", group=grp_trend)

// Colors Settings
grp_color       = "Cài đặt màu sắc Boxes"
colorBox21      = input.color(color.new(color.navy, 40), "Màu viền Box 21", group=grp_color)
bgBox21         = input.color(color.new(color.navy, 95), "Màu nền Box 21", group=grp_color)
colorBox9       = input.color(color.new(color.teal, 60), "Màu viền Box 9", group=grp_color)
bgBox9          = input.color(color.new(color.teal, 95), "Màu nền Box 9", group=grp_color)
colorVCP        = input.color(color.new(#00bcd4, 45), "Màu viền VCP Nén", group=grp_color)
bgVCP           = input.color(color.new(#00bcd4, 85), "Màu nền VCP Nén", group=grp_color)

// --- [2] TÍNH TOÁN CƠ BẢN ---
ema21   = ta.ema(close, emaLen)
ma9     = ta.sma(close, ma9Len)
avgVol  = ta.sma(volume, 50)
atr     = ta.atr(14)

plot(ema21, color=color.new(color.orange, 50), title="EMA 21")
plot(ma9, color=color.new(color.teal, 50), title="MA 9")

rawLookback = useAutoLookback ? math.max(20, math.round(ta.correlation(close, ema21, 20) * 40)) : manualLookback
lookback    = int(nz(rawLookback, manualLookback))

highLook = ta.highest(high, lookback)
lowLook  = ta.lowest(low, lookback)

bodyHigh = math.max(open, close), bodyLow = math.min(open, close)
high21 = ta.highest(bodyHigh, 21), low21 = ta.lowest(bodyLow, 21)
high9  = ta.highest(bodyHigh, 9),  low9  = ta.lowest(bodyLow, 9)

[plus, minus, adxValue] = ta.dmi(14, 14)
isSideway = adxValue < adxThreshold and (highLook - lowLook) < (atr * 6)
isVCP = (high21 - low21) > 0 and ((high9 - low9) / (high21 - low21)) <= vcpRatio

// --- [3] HỆ THỐNG BOXES VCP (FIXED SYNTAX) ---
var box box21 = box.new(na, na, na, na, border_color=colorBox21, bgcolor=bgBox21, border_style=line.style_dashed) 
var box box9  = box.new(na, na, na, na)
var label vcpLab = label.new(na, na, "", color=colorVCP, textcolor=color.white, style=label.style_label_down, size=size.small)

if barstate.islast
    if not na(high21) and not na(low21)
        box.set_lefttop(box21, bar_index - 21, high21)
        box.set_rightbottom(box21, bar_index + offset, low21)
    
    if not na(high9) and not na(low9)
        box.set_lefttop(box9, bar_index - 9, high9)
        box.set_rightbottom(box9, bar_index + offset, low9)
        box.set_border_color(box9, isVCP ? colorVCP : colorBox9)
        box.set_bgcolor(box9, isVCP ? bgVCP : bgBox9)

    if isVCP and not na(high9)
        label.set_xy(vcpLab, bar_index + (offset / 2), high9)
        label.set_text(vcpLab, "VCP Nén")
    else
        label.set_xy(vcpLab, na, na)

// --- [4] TRENDLINES (FIXED DOWNWARD LOGIC) ---

// 1. Trendline Chính (Đường Đỏ - Kháng cự dài)
ph = ta.pivothigh(high, pivotLen, pivotLen)
var line resLine = line.new(na, na, na, na, color=resColor, width=2, extend=extend.right)
if not na(ph) and showTrend
    int p_idx = 0, float p_val = 0.0
    for i = 1 to 200
        if not na(ph[i]) and ph[i] > ph // Đỉnh trước cao hơn đỉnh sau = dốc xuống
            p_idx := bar_index[i + pivotLen], p_val := ph[i]
            break
    if p_val > 0
        line.set_xy1(resLine, p_idx, p_val), line.set_xy2(resLine, bar_index[pivotLen], ph)

// 2. Trendline Hỗ trợ (Đường Xanh - Nối đáy dốc lên)
pl = ta.pivotlow(low, pivotLen, pivotLen)
var line supLine = line.new(na, na, na, na, color=supColor, width=2, extend=extend.right)
if not na(pl) and showTrend
    int p_idx = 0, float p_val = 0.0
    for i = 1 to 200
        if not na(pl[i]) and pl[i] < pl // Đáy trước thấp hơn đáy sau = dốc lên
            p_idx := bar_index[i + pivotLen], p_val := pl[i]
            break
    if p_val > 0
        line.set_xy1(supLine, p_idx, p_val), line.set_xy2(supLine, bar_index[pivotLen], pl)

// 3. Trendline Ngắn hạn (Đường Tím - Kháng cự ngắn hạn hướng xuống)
phShort = ta.pivothigh(high, pivotLenShort, pivotLenShort)
var line resLineShort = line.new(na, na, na, na, color=resColorShort, width=2, extend=extend.right)

if not na(phShort) and showTrendShort
    int ps_idx = 0, float ps_val = 0.0
    for i = 1 to 100
        if not na(phShort[i]) and phShort[i] > phShort // Ép logic hướng xuống
            ps_idx := bar_index[i + pivotLenShort], ps_val := phShort[i]
            break
    if ps_val > 0
        line.set_xy1(resLineShort, ps_idx, ps_val), line.set_xy2(resLineShort, bar_index[pivotLenShort], phShort)

// --- [5] WYCKOFF ---
var float lockedHigh = na, var float lockedLow = na
var bool scTriggered = false, var bool bcTriggered = false

if isSideway and not isSideway[1]
    lockedHigh := highLook, lockedLow := lowLook
    scTriggered := false, bcTriggered := false

volConfirm = volume > avgVol * volSensitivity
if showWyckoff and isSideway and not na(lockedHigh)
    if volConfirm
        if not scTriggered and low <= lockedLow
            label.new(bar_index, low, "SC", yloc=yloc.belowbar, color=color.red, textcolor=color.white, style=label.style_label_up, size=size.small)
            scTriggered := true
        if not bcTriggered and high >= lockedHigh
            label.new(bar_index, high, "BC", yloc=yloc.abovebar, color=color.green, textcolor=color.white, style=label.style_label_down, size=size.small)
            bcTriggered := true

// --- [6] TREND & BREAKOUT SIGNALS ---
buy_cross = ta.crossover(ma9, ema21)
sell_cross = ta.crossunder(ma9, ema21)
buy_break = ta.crossover(close, isSideway[1] ? lockedHigh : highLook[1])
sell_break = ta.crossunder(close, isSideway[1] ? lockedLow : lowLook[1])

plotshape(buy_cross, title="Buy Cross", style=shape.triangleup, location=location.belowbar, color=color.new(color.green, 20), size=size.huge)
plotshape(sell_cross, title="Sell Cross", style=shape.triangledown, location=location.abovebar, color=color.new(color.red, 20), size=size.huge)

if buy_break
    label.new(bar_index, low, "BUY Break", yloc=yloc.belowbar, color=color.new(color.green, 40), textcolor=color.white, style=label.style_label_up, size=size.small)
if sell_break
    label.new(bar_index, high, "SELL Break", yloc=yloc.abovebar, color=color.new(color.red, 40), textcolor=color.white, style=label.style_label_down, size=size.small)

// --- [7] FVG (ICT) ---
var box[] bullFVGs = array.new_box()
var float[] bullTops = array.new_float()
var box[] bearFVGs = array.new_box()
var float[] bearBots = array.new_float()

if showFVG and barstate.isconfirmed
    if low > high[2] and close[1] > open[1] and not na(high[2])
        array.push(bullFVGs, box.new(bar_index-2, high[2], bar_index+offset, low, bgcolor=color.new(color.teal, 90), border_color=na))
        array.push(bullTops, high[2])
    if high < low[2] and close[1] < open[1] and not na(low[2])
        array.push(bearFVGs, box.new(bar_index-2, low[2], bar_index+offset, high, bgcolor=color.new(color.red, 90), border_color=na))
        array.push(bearBots, low[2])

if array.size(bullFVGs) > 0
    for i = array.size(bullFVGs) - 1 to 0
        bx = array.get(bullFVGs, i), t = array.get(bullTops, i)
        box.set_right(bx, bar_index + offset)
        if (not na(low) and low <= t) or array.size(bullFVGs) > maxFVGCount
            box.delete(bx), array.remove(bullFVGs, i), array.remove(bullTops, i)

if array.size(bearFVGs) > 0
    for i = array.size(bearFVGs) - 1 to 0
        bx = array.get(bearFVGs, i), b = array.get(bearBots, i)
        box.set_right(bx, bar_index + offset)
        if (not na(high) and high >= b) or array.size(bearFVGs) > maxFVGCount
            box.delete(bx), array.remove(bearFVGs, i), array.remove(bearBots, i)
